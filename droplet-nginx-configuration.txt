sudo nano /etc/nginx/sites-available/globalcitystreaming.com

server {
    listen 80;
    server_name globalcitystreaming.com www.globalcitystreaming.com;

    location / {
        proxy_pass http://localhost:3000; # Your Docker app port
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}


sudo ln -s /etc/nginx/sites-available/globalcitystreaming.com /etc/nginx/sites-enabled/

sudo nginx -t

sudo systemctl reload nginx
